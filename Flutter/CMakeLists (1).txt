// TRANSLATION NOTES:
//   • darkColorScheme(…)  →  ColorScheme.dark(…)  — same Material-3 fields
//   • SideEffect { window.statusBarColor }  →  desktop has no status/nav bars;
//     the equivalent is handled by the OS window title-bar; no action needed.
//   • MaterialTheme { colorScheme, typography, shapes }  →  identical in Flutter
//   • Typography TextStyle sp  →  Flutter TextStyle fontSize in logical pixels;
//     'sp' == logical pixel on desktop (no font scaling from accessibility)

import 'package:flutter/material.dart';
import 'app_colors.dart';

// ── Color scheme (mirrors GDarkColorScheme) ───────────────────────────────
final gColorScheme = ColorScheme.dark(
  primary: kSapphire,
  onPrimary: kOnSapphire,
  primaryContainer: kSapphireContainer,
  onPrimaryContainer: kSky,
  secondary: kSky,
  onSecondary: kBlack,
  secondaryContainer: kNavy700,
  onSecondaryContainer: kSky,
  tertiary: kGradeA,
  onTertiary: kBlack,
  tertiaryContainer: kGradeAContainer,
  onTertiaryContainer: kGradeA,
  error: kError,
  onError: kBlack,
  errorContainer: kErrorContainer,
  onErrorContainer: kError,
  surface: kSurf2,
  onSurface: kTextPrimary,
  surfaceContainerHighest: kSurf3,
  onSurfaceVariant: kTextSecondary,
  outline: kNavyOutline,
  outlineVariant: kNavyDivider,
  scrim: kBlack,
  inverseSurface: kTextPrimary,
  onInverseSurface: kSurf2,
  inversePrimary: kSapphireDim,
  surfaceTint: kSapphire,
);

// ── Typography (mirrors GTypography) ─────────────────────────────────────
final gTextTheme = const TextTheme(
  displayLarge:  TextStyle(fontWeight: FontWeight.bold,      fontSize: 57, height: 64/57),
  displayMedium: TextStyle(fontWeight: FontWeight.w600,      fontSize: 45, height: 52/45),
  headlineLarge: TextStyle(fontWeight: FontWeight.bold,      fontSize: 32, height: 40/32),
  headlineMedium:TextStyle(fontWeight: FontWeight.w600,      fontSize: 28, height: 36/28),
  headlineSmall: TextStyle(fontWeight: FontWeight.w600,      fontSize: 24, height: 32/24),
  titleLarge:    TextStyle(fontWeight: FontWeight.w600,      fontSize: 22, height: 28/22),
  titleMedium:   TextStyle(fontWeight: FontWeight.w500,      fontSize: 16, height: 24/16, letterSpacing: 0.15),
  titleSmall:    TextStyle(fontWeight: FontWeight.w500,      fontSize: 14, height: 20/14, letterSpacing: 0.1),
  bodyLarge:     TextStyle(fontWeight: FontWeight.normal,    fontSize: 16, height: 24/16, letterSpacing: 0.5),
  bodyMedium:    TextStyle(fontWeight: FontWeight.normal,    fontSize: 14, height: 20/14, letterSpacing: 0.25),
  bodySmall:     TextStyle(fontWeight: FontWeight.normal,    fontSize: 12, height: 16/12, letterSpacing: 0.4),
  labelLarge:    TextStyle(fontWeight: FontWeight.w600,      fontSize: 14, height: 20/14, letterSpacing: 0.1),
  labelMedium:   TextStyle(fontWeight: FontWeight.w500,      fontSize: 12, height: 16/12, letterSpacing: 0.5),
  labelSmall:    TextStyle(fontWeight: FontWeight.w500,      fontSize: 11, height: 16/11, letterSpacing: 0.5),
);

// ── Shapes (mirrors GShapes) ──────────────────────────────────────────────
// Flutter ThemeData.useMaterial3 uses shape overrides per-widget;
// we replicate the Kotlin Shapes object as named BorderRadius constants.
const kRadiusXs = BorderRadius.all(Radius.circular(4));
const kRadiusSm = BorderRadius.all(Radius.circular(8));
const kRadiusMd = BorderRadius.all(Radius.circular(12));
const kRadiusLg = BorderRadius.all(Radius.circular(16));
const kRadiusXl = BorderRadius.all(Radius.circular(22));

// ── ThemeData ─────────────────────────────────────────────────────────────
ThemeData buildGTheme() {
  return ThemeData(
    useMaterial3: true,
    colorScheme: gColorScheme,
    textTheme: gTextTheme.apply(
      bodyColor: kTextPrimary,
      displayColor: kTextPrimary,
    ),
    scaffoldBackgroundColor: kBlack,
    cardColor: kSurf2,
    dividerColor: kNavyDivider,
    // Navigation bar (bottom nav) theme
    navigationBarTheme: NavigationBarThemeData(
      backgroundColor: kNavy900,
      elevation: 0,
      indicatorColor: kSapphire.withOpacity(0.12),
      iconTheme: WidgetStateProperty.resolveWith((states) {
        if (states.contains(WidgetState.selected)) {
          return const IconThemeData(color: kSapphire);
        }
        return const IconThemeData(color: kTextDisabled);
      }),
      labelTextStyle: WidgetStateProperty.resolveWith((states) {
        final color = states.contains(WidgetState.selected)
            ? kSapphire
            : kTextDisabled;
        return TextStyle(
          fontSize: 10,
          color: color,
          fontWeight: FontWeight.w500,
        );
      }),
    ),
    // AppBar theme
    appBarTheme: const AppBarTheme(
      backgroundColor: kNavy900,
      foregroundColor: kTextPrimary,
      elevation: 0,
      surfaceTintColor: Colors.transparent,
    ),
    // Outlined text field
    inputDecorationTheme: InputDecorationTheme(
      filled: true,
      fillColor: kSurf2,
      border: OutlineInputBorder(
        borderRadius: BorderRadius.circular(12),
        borderSide: const BorderSide(color: kNavyOutline),
      ),
      enabledBorder: OutlineInputBorder(
        borderRadius: BorderRadius.circular(12),
        borderSide: const BorderSide(color: kNavyOutline),
      ),
      focusedBorder: OutlineInputBorder(
        borderRadius: BorderRadius.circular(12),
        borderSide: const BorderSide(color: kSapphire, width: 2),
      ),
      hintStyle: const TextStyle(color: kTextSecondary),
    ),
    // Slider
    sliderTheme: const SliderThemeData(
      activeTrackColor: kSapphire,
      thumbColor: kSapphire,
      inactiveTrackColor: kNavy600,
      overlayColor: Color(0x1A2979FF),
    ),
    // Switch
    switchTheme: SwitchThemeData(
      thumbColor: WidgetStateProperty.resolveWith((states) =>
          states.contains(WidgetState.selected) ? kBlack : kTextSecondary),
      trackColor: WidgetStateProperty.resolveWith((states) =>
          states.contains(WidgetState.selected) ? kSapphire : kNavy500),
    ),
    // Snackbar
    snackBarTheme: SnackBarThemeData(
      backgroundColor: kSurf3,
      contentTextStyle: const TextStyle(color: kTextPrimary),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
      behavior: SnackBarBehavior.floating,
    ),
    // Card
    cardTheme: CardTheme(
      color: kSurf2,
      elevation: 0,
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(14)),
    ),
    // Elevated button (primary)
    elevatedButtonTheme: ElevatedButtonThemeData(
      style: ElevatedButton.styleFrom(
        backgroundColor: kSapphire,
        foregroundColor: kOnSapphire,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
      ),
    ),
    // Outlined button
    outlinedButtonTheme: OutlinedButtonThemeData(
      style: OutlinedButton.styleFrom(
        foregroundColor: kTextSecondary,
        side: const BorderSide(color: kNavyOutline),
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
      ),
    ),
    // FAB
    floatingActionButtonTheme: const FloatingActionButtonThemeData(
      backgroundColor: kSapphire,
      foregroundColor: kOnSapphire,
    ),
    // Progress indicators
    progressIndicatorTheme: const ProgressIndicatorThemeData(
      color: kSapphire,
      linearTrackColor: kNavy600,
      circularTrackColor: kNavy600,
    ),
  );
}
